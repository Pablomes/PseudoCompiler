FUNCTION IsPrime(n:INTEGER) RETURNS BOOLEAN
IF n <= 1 THEN
RETURN FALSE
ENDIF
FOR i <- 2 TO INT(n^0.5)
IF n MOD i = 0 THEN
RETURN FALSE
ENDIF
NEXT i
RETURN TRUE
ENDFUNCTION

DECLARE Limit:INTEGER
OUTPUT "Enter the upper limit for prime numbers:"
INPUT Limit
OUTPUT "Prime numbers up to ", Limit, ":"
FOR num <- 2 TO Limit
IF IsPrime(num) THEN
OUTPUT num
ENDIF
NEXT num
